        function initNavigation() {
            // Навигация по страницам
            document.querySelectorAll('.nav-link').forEach(link => {
                link.addEventListener('click', function(e) {
                    e.preventDefault();
                    const page = this.dataset.page;
                    
                    // Обновить активные ссылки
                    document.querySelectorAll('.nav-link').forEach(l => l.classList.remove('active'));
                    this.classList.add('active');
                    
                    // Показать выбранную страницу
                    document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
                    document.getElementById(page).classList.add('active');
                    
                    // Скрыть мобильное меню
                    if (window.innerWidth <= 768) {
                        document.querySelector('nav ul').style.display = 'none';
                    }
                });
            });

            // Мобильное меню
            document.querySelector('.mobile-menu-btn').addEventListener('click', function() {
                const nav = document.querySelector('nav ul');
                nav.style.display = nav.style.display === 'flex' ? 'none' : 'flex';
            });
        }

                    const cards = document.querySelectorAll('#classes-grid .card');
                    
                    cards.forEach(card => {
                        const cardClass = classes.find(c => c.name === card.querySelector('h3').textContent);
                        if (group === 'all' || cardClass.group === group) {
                            card.style.display = 'block';

                        }
                    });
                });
            });

        // Поиск
        function initSearch() {
            document.getElementById('search-btn').addEventListener('click', performSearch);
            document.getElementById('search-input').addEventListener('keypress', function(e) {
                if (e.key === 'Enter') performSearch();
            });
        }

        function performSearch() {
            const query = document.getElementById('search-input').value.toLowerCase().trim();
            if (!query) return;
            
            let results = [];
            Object.values(materials).forEach(category => {
                category.forEach(item => {
                    if (item.title.toLowerCase().includes(query) || 
                        item.author.toLowerCase().includes(query) ||
                        item.class.toLowerCase().includes(query) ||
                        item.subject.toLowerCase().includes(query)) {
                        results.push(item);
                    }
                });
            });
            
            if (results.length > 0) {
                alert(`Найдено материалов: ${results.length}\n\nПоказаны первые 5 результатов:\n\n${
                    results.slice(0,5).map(item => 
                        `• ${item.title}\n  ${item.author} | ${item.class} | ${item.subject}`
                    ).join('\n\n')
                }`);
            } else {
                alert('Ничего не найдено');
            }
            
            document.getElementById('search-input').value = '';
        }

        // Просмотр материала
        function viewMaterial(id) {
            let item = null;
            for (const category in materials) {
                item = materials[category].find(m => m.id === id);
                if (item) break;
            }
            
            if (item) {
                alert(`Материал: ${item.title}\n\nАвтор: ${item.author}\nКласс: ${item.class}\nПредмет: ${item.subject}\n\nВ реальном приложении здесь будет открыт PDF-файл с материалом.`);
            }
        }

        // ИИ помощник
        function initAI() {
            const messages = document.getElementById('ai-messages');
            const input = document.getElementById('ai-input');
            const sendBtn = document.getElementById('ai-send');
            
            function addMessage(text, sender) {
                const message = document.createElement('div');
                message.className = `message ${sender}-message`;
                message.innerHTML = `<div class="message-content">${text}</div>`;
                messages.appendChild(message);
                messages.scrollTop = messages.scrollHeight;
            }
            
            function sendMessage() {
                const text = input.value.trim();
                if (!text) return;
                
                addMessage(text, 'user');
                input.value = '';
                
                // Имитация ответа ИИ
                setTimeout(() => {
                    let response = '';
                    
                    if (text.toLowerCase().includes('привет')) {
                        response = "Привет! Я ИИ-помощник по учебным материалам. Могу помочь с любыми вопросами по школьной программе.";
                    } else if (text.toLowerCase().includes('спасибо')) {
                        response = "Пожалуйста! Обращайтесь, если будут еще вопросы.";
                    } else if (text.toLowerCase().includes('математик') || text.toLowerCase().includes('алгебр') || text.toLowerCase().includes('геометри')) {
                        response = "По математике могу помочь: объяснить тему, решить задачу, проверить решение. Уточните, что именно нужно?";
                    } else if (text.toLowerCase().includes('русск') || text.toLowerCase().includes('язык')) {
                        response = "По русскому языку могу: проверить упражнение, объяснить правило, помочь с сочинением. Что вас интересует?";
                    } else if (text.toLowerCase().includes('физик')) {
                        response = "По физике могу объяснить законы, помочь решить задачи, разобрать лабораторные работы.";
                    } else if (text.toLowerCase().includes('хими')) {
                        response = "По химии могу: объяснить реакции, помочь с уравнениями, разобрать задачи.";
                    } else {
                        response = "Я могу помочь с учебными вопросами по любому предмету. Задайте конкретный вопрос, и я постараюсь дать развернутый ответ.";
                    }
                    
                    addMessage(response, 'ai');
                }, 1000);
            }
            
            sendBtn.addEventListener('click', sendMessage);
            input.addEventListener('keypress', (e) => {
                if (e.key === 'Enter') sendMessage();
            });
        }
